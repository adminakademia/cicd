### Instalacja Docker w systemie ###

# Add Docker's official GPG key:
sudo apt update
sudo apt install ca-certificates curl
sudo install -m 0755 -d /etc/apt/keyrings
sudo curl -fsSL https://download.docker.com/linux/debian/gpg -o /etc/apt/keyrings/docker.asc
sudo chmod a+r /etc/apt/keyrings/docker.asc

# Add the repository to Apt sources:
sudo tee /etc/apt/sources.list.d/docker.sources <<EOF
Types: deb
URIs: https://download.docker.com/linux/debian
Suites: $(. /etc/os-release && echo "$VERSION_CODENAME")
Components: stable
Signed-By: /etc/apt/keyrings/docker.asc
EOF


sudo apt update
sudo apt install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin


### Instalacja gitlab srv jako kontener Docker ###

# Utwórz katalog główny dla projektu
mkdir -p /gitlabsrv

# Utwórz podkatalogi dla wolumenów
mkdir -p /gitlabsrv/{config,data,logs}

# Przejdź do katalogu
cd /gitlabsrv


# Utwórz plik docker-compose.yml w katalogu /opt/gitlab-docker:

mcedit /gitlabsrv/docker-compose.yml

# z zawartością:


services:
  gitlab:
    image: gitlab/gitlab-ce:latest
    container_name: gitlab
    restart: always
    hostname: 'gitlab.mojadomena.pl'
    environment:
      GITLAB_OMNIBUS_CONFIG: |
        external_url 'http://gitlab.mojadomena.pl'
        gitlab_rails['gitlab_shell_ssh_port'] = 22
        gitlab_rails['time_zone'] = 'Europe/Warsaw'
        prometheus_monitoring['enable'] = false
        puma['worker_processes'] = 2
        puma['min_threads'] = 1
        puma['max_threads'] = 4
        sidekiq['concurrency'] = 5
        registry['enable'] = false
    ports:
      - '80:80'
      - '443:443'
      - '22:22'
    volumes:
      - './config:/etc/gitlab'
      - './logs:/var/log/gitlab'
      - './data:/var/opt/gitlab'
    shm_size: '256m'


# poniżej z ciut większymi wyjaśnieniami to samo co wyżej:
services:
  gitlab:
    image: gitlab/gitlab-ce:latest
    container_name: gitlab
    restart: always
    hostname: 'gitlab.mojadomena.pl' # Zmień na swoją domenę lub IP
    environment:
      GITLAB_OMNIBUS_CONFIG: |
        # Podstawowa konfiguracja URL
        external_url 'http://gitlab.mojadomena.pl' # Użyj https jeśli masz certyfikaty
        
        # Konfiguracja SSH (jeśli zmieniasz port, patrz niżej)
        gitlab_rails['gitlab_shell_ssh_port'] = 22

        # Ustawienie strefy czasowej
        gitlab_rails['time_zone'] = 'Europe/Warsaw'
        
        # OPTYMALIZACJA PAMIĘCI (Dla serwerów z <8GB RAM) [4, 8]
        # Wyłączenie monitoringu Prometheus (oszczędza ok. 200-500MB RAM)
        prometheus_monitoring['enable'] = false
        
        # Redukcja procesów Puma (serwer WWW)
        puma['worker_processes'] = 2
        puma['min_threads'] = 1
        puma['max_threads'] = 4
        
        # Redukcja procesów Sidekiq (kolejkowanie zadań)
        sidekiq['concurrency'] = 5
        
        # Wyłączenie rejestru kontenerów (jeśli nie planujesz używać)
        registry['enable'] = false
        
    ports:
      - '80:80'
      - '443:443'
      - '22:22' 
    volumes:
      - './config:/etc/gitlab'
      - './logs:/var/log/gitlab'
      - './data:/var/opt/gitlab'
    shm_size: '256m' 
	

###
#• shm_size: '256m': To ustawienie jest krytyczne. Domyślny rozmiar pamięci współdzielonej w Dockerze #(64MB) jest zbyt mały dla usług monitoringu GitLaba (Prometheus), co często powoduje błędy. Zwiększenie # do 256MB rozwiązuje ten problem
###

##### Dodajemy runnera ##### 



# 1. Dodanie Runnera do docker-compose.yml
# Najpierw musisz dodać usługę runnera do swojego istniejącego pliku. Kluczowe jest zamontowanie gniazda # Dockera (/var/run/docker.sock), aby runner mógł uruchamiać kontenery na Twoim serwerze.


# Zaktualizuj swój plik o poniższą sekcję:

mcedit /gitlabsrv/docker-compose.yml


services:
  gitlab:
    # Twoja obecna konfiguracja...

  gitlab-runner:
    image: gitlab/gitlab-runner:latest
    container_name: gitlab-runner
    restart: always
    volumes:
      - '/var/run/docker.sock:/var/run/docker.sock' # Dostęp do Dockera hosta
      - './runner-config:/etc/gitlab-runner'      # Trwałość konfiguracji

# Następnie uruchom go: 

cd /gitlabsrv/
docker compose up -d --force-recreate


# 2. Pobranie tokenu rejestracyjnego
# Zanim zarejestrujesz runnera, potrzebujesz tokenu ze swojego interfejsu GitLab:

# Zaloguj się jako administrator (root).
# Przejdź do Admin Area > CI/CD > Runners.
# Kliknij Create instance runner.
# Zdefiniuj tagi (np. docker) i kliknij Create runner. Zobaczysz token – skopiuj go.


# 3. Rejestracja Runnera

cp /gitlabsrv/config/ssl/certwild.pem /gitlabsrv/runner-config

# Teraz musisz "połączyć" kontener runnera z Twoim serwerem GitLab. Wykonaj poniższą komendę w terminalu # serwera:

docker exec -it gitlab-runner gitlab-runner register \
  --non-interactive \
  --url "https://gitlab.mojadomena.pl/" \
  --token "<tutaj_wpisz_token>" \
  --tls-ca-file "/etc/gitlab-runner/certwild.pem" \
  --executor "docker" \
  --docker-image "alpine:latest" \
  --description "docker-runner"



# 4. Ważna konfiguracja (Dind - Docker in Docker)
# Aby móc wewnątrz potoku budować własne obrazy (używać komendy docker build), musisz edytować plik # konfiguracyjny runnera, który utworzył się w ./runner-config/config.toml:

# Otwórz plik: 
mcedit /gitlabsrv/runner-config/config.toml

# W sekcji [runners.docker] znajdź parametr privileged i zmień go na true.
# W sekcji volumes upewnij się, że masz: volumes = ["/var/run/docker.sock:/var/run/docker.sock", "/cache"]
